import express from 'express';
import helmet from 'helmet';
import cors from 'cors';
import dotenv from 'dotenv';
import path from 'path';
import { fileURLToPath } from 'url';
import db from './db.js';

dotenv.config();
const app = express();
app.use(helmet());
app.use(cors());
app.use(express.json());

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);
const PORT = process.env.PORT || 3000;

// Trucks
app.get('/api/trucks', (req, res) => {
  const rows = db.prepare('SELECT * FROM trucks ORDER BY created_at DESC').all();
  res.json(rows);
});
app.post('/api/trucks', (req, res) => {
  const { name, cuisine, description, lat, lng, hours, social } = req.body;
  const stmt = db.prepare(`INSERT INTO trucks (name, cuisine, description, lat, lng, hours, social)
                           VALUES (?, ?, ?, ?, ?, ?, ?)`);
  const info = stmt.run(name, cuisine, description, lat, lng, hours, JSON.stringify(social || {}));
  const row = db.prepare('SELECT * FROM trucks WHERE id = ?').get(info.lastInsertRowid);
  res.status(201).json(row);
});

// Events
app.get('/api/events', (req, res) => {
  const rows = db.prepare('SELECT * FROM events ORDER BY date ASC, created_at DESC').all();
  res.json(rows);
});
app.post('/api/events', (req, res) => {
  const { title, date, location, expected_headcount, cuisines_needed, trucks_needed, notes } = req.body;
  const stmt = db.prepare(`INSERT INTO events (title, date, location, expected_headcount, cuisines_needed, trucks_needed, notes)
                           VALUES (?, ?, ?, ?, ?, ?, ?)`);
  const info = stmt.run(title, date, location, expected_headcount, cuisines_needed, trucks_needed, notes);
  const row = db.prepare('SELECT * FROM events WHERE id = ?').get(info.lastInsertRowid);
  res.status(201).json(row);
});

// Serve client build (prod)
const clientDist = path.join(__dirname, '..', 'client', 'dist');
app.use(express.static(clientDist));
app.get('*', (req, res) => {
  try {
    res.sendFile(path.join(clientDist, 'index.html'));
  } catch {
    res.status(200).json({ ok: true, message: 'API running. Build the client to serve UI here.' });
  }
});

app.listen(PORT, () => console.log(`Server http://localhost:${PORT}`));
